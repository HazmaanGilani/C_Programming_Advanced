Algorithm for Convertion Infix to Postfix Expression:
opstk = the empty stack;

 while (not end of input) {

 symbol = next input character;

 if (symbol is an operand) 

 add symbol to postfix string

 else {

 while (!empty (opstk) and prcd (top (opstk),
symbol)>0){

 topsymb = pop(opstk);

 add topsymb to the postfix string; 

 }/*end while*/

 if (empty (opstk) || symbol! = ')' ) 

 push (opstk,symbol); 

else 
/*pop the open parenthesis and discard it */

topsymb = pop(opstk);

} /* end else */

}/* end while */

/* output any remaining operator */

while (!empty (opstk)){

 top symbol = pop (opstk);

 add topsymb to the postfix string;

} /* end while * /
